62 /*
63  * Moves the cursor to the specified location. If row is 0, the text will appear on the first line
64  * else is will appear on the second line. Col is a number between 0 and 15 and represents where
65  * the text will begin along the line.
66  */
67 void move_cursor(uint8_t row , uint8_t col){
68 
69  uint8_t loc = col | (row << 6) | (1 << 7);
70 
71  SPDR = 0x00; //command, not data
72  _delay_loop_1(9); //wait 2us for data to leave SPI port
73  SPDR = loc; // cursor go to location
74  _delay_loop_1(9); //wait 2us for data to leave SPI port
75  strobe_lcd();
76  delay_ms1(2);
77 }

99 void char2lcd(char a_char){
100  SPDR = 0x01; //set SR for data xfer with LSB=1

102  SPDR = a_char; //send the char to the SPI port
103  _delay_loop_1(9); //wait 2us for data to leave SPI port

120  int count;

122  SPDR = 0x01; //set SR for data
123  _delay_loop_1(9); //wait 2us for data to leave SPI port

125  _delay_loop_1(9); //wait 2us for data to leave SPI port
126  strobe_lcd();
127  delay_ms1(1);
